CREATE PROCEDURE PROCEDURE_3(
    DATE_BEG DMN_DATE,
    DATE_END DMN_DATE)
RETURNS (
    ID_SUBJECT DMN_PK,
    NAME DMN_STR,
    AUCTION_DATE DMN_DATE,
    AUCTION_PLACE DMN_STR)
AS
begin
  for
    select s.id_subject, s.subject_name, a.start_date, a.start_place
        from auction a inner join subject_auction sa
            on sa.id_auction = a.id_auction
            inner join subject s
            on s.id_subject = sa.id_subject
        where a.start_date > :date_beg and a.start_date < :date_end
        group by s.id_subject, s.subject_name, a.start_date, a.start_place
    into :id_subject, :name, :auction_date, :auction_place
  do suspend;
end
