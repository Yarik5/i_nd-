CREATE PROCEDURE PROCEDURE_1(
    DATE_BEG DMN_DATE,
    DATE_END DMN_DATE)
RETURNS (
    ID_AUCTION DMN_PK,
    PLACE DMN_STR,
    AUCTION_DATE DMN_DATE,
    SNAME DMN_STR)
AS
begin
  for
    select a.id_auction, a.start_place, a.start_date, s.subject_name
        from auction a inner join subject_auction sa
            on sa.id_auction = a.id_auction
            inner join subject s
            on s.id_subject = sa.id_subject
        where a.start_date > :date_beg and a.start_date < :date_end
        order by a.start_date
    into :id_auction, :place, :auction_date, :sname
  do suspend;
end
